def execute(message, rolls, sides, modifier)
  dice_roll = IRCBot::DiceRoll::Generator.new(message,
    sides: sides.to_i,
    rolls: rolls.to_i,
    modifier: (modifier ? modifier.delete(' ').to_i : 0)
  )
  
  dice_roll.run

  print_total(message, dice_roll)
  print_dice(message, dice_roll)
end
